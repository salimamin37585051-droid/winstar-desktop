1
Create free Supabase account
Go to supabase.com → Sign Up free → New Project → name it winstar → wait ~2 min for setup
2
Create database tables
In Supabase → SQL Editor → paste and run this SQL:
create table ws_users(id serial primary key,username text unique,password text,name text,role text,branch text,initials text,last_login text,active boolean default true);
create table ws_guards(id serial primary key,gid text,name text,branch text,site text,pos text,phone text,status text,joined text);
create table ws_items(id serial primary key,name text,cat text,size text,branch text,total int,disp int default 0,ret int default 0,price numeric);
create table ws_dispatches(id serial primary key,ref text,gid int,date text,branch text,items_json text,by_name text,notes text);
create table ws_returns(id serial primary key,ref text,gid int,date text,branch text,dref text,items_json text,cond text,notes text);
create table ws_requests(id serial primary key,ref text,gid int,date text,item text,reason text,priority text,status text);
create table ws_branches(id serial primary key,name text,loc text,mgr text,status text);
create table ws_audit(id serial primary key,ts text,user_name text,msg text,type text);
alter table ws_users enable row level security;
alter table ws_guards enable row level security;
alter table ws_items enable row level security;
alter table ws_dispatches enable row level security;
alter table ws_returns enable row level security;
alter table ws_requests enable row level security;
alter table ws_branches enable row level security;
alter table ws_audit enable row level security;
create policy "allow all" on ws_users for all using (true) with check (true);
create policy "allow all" on ws_guards for all using (true) with check (true);
create policy "allow all" on ws_items for all using (true) with check (true);
create policy "allow all" on ws_dispatches for all using (true) with check (true);
create policy "allow all" on ws_returns for all using (true) with check (true);

create policy "allow all" on ws_branches for all using (true) with check (true);
create policy "allow all" on ws_audit for all using (true) with check (true);
