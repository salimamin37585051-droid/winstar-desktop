-- WINSTAR SECURE SETUP (fresh install) create table ws_users(id serial primary key,username text unique,password text,name text,role text,branch text,initials text,last_login text,active boolean default true); create table ws_guards(id serial primary key,gid text,name text,branch text,site text,pos text,phone text,status text,joined text); create table ws_items(id serial primary key,name text,cat text,size text,branch text,total int default 0,disp int default 0,ret int default 0,price numeric default 0); create table ws_dispatches(id serial primary key,ref text,gid int,date text,branch text,items_json text,by_name text,notes text); create table ws_returns(id serial primary key,ref text,gid int,date text,branch text,dref text,items_json text,cond text,notes text); create table ws_requests(id serial primary key,ref text,gid int,date text,item text,reason text,priority text,status text default 'pending'); create table ws_branches(id serial primary key,name text,loc text,mgr text,status text default 'active'); create table ws_audit(id serial primary key,ts text,user_name text,msg text,type text); alter table ws_users enable row level security; alter table ws_guards enable row level security; alter table ws_items enable row level security; alter table ws_dispatches enable row level security; alter table ws_returns enable row level security; alter table ws_requests enable row level security; alter table ws_branches enable row level security; alter table ws_audit enable row level security; -- Read: anon key can SELECT (app needs this to load data after login) create policy "ws_read" on ws_users for select using (true); create policy "ws_read" on ws_guards for select using (true); create policy "ws_read" on ws_items for select using (true); create policy "ws_read" on ws_dispatches for select using (true); create policy "ws_read" on ws_returns for select using (true); create policy "ws_read" on ws_requests for select using (true); create policy "ws_read" on ws_branches for select using (true); create policy "ws_read" on ws_audit for select using (true); -- Write: anon key can INSERT/UPDATE (app needs this to save records) create policy "ws_insert" on ws_users for insert with check (true); create policy "ws_insert" on ws_guards for insert with check (true); create policy "ws_insert" on ws_items for insert with check (true); create policy "ws_insert" on ws_dispatches for insert with check (true); create policy "ws_insert" on ws_returns for insert with check (true); create policy "ws_insert" on ws_requests for insert with check (true); create policy "ws_insert" on ws_branches for insert with check (true); create policy "ws_insert" on ws_audit for insert with check (true); create policy "ws_update" on ws_users for update using (true) with check (true); create policy "ws_update" on ws_guards for update using (true) with check (true); create policy "ws_update" on ws_items for update using (true) with check (true); create policy "ws_update" on ws_requests for update using (true) with check (true); create policy "ws_update" on ws_branches for update using (true) with check (true); -- Delete: restricted to specific tables only (matches app's delete features) create policy "ws_delete" on ws_users for delete using (true); create policy "ws_delete" on ws_guards for delete using (true); create policy "ws_delete" on ws_items for delete using (true); create policy "ws_delete" on ws_dispatches for delete using (true); create policy "ws_delete" on ws_returns for delete using (true); create policy "ws_delete" on ws_requests for delete using (true); create policy "ws_delete" on ws_branches for delete using (true); -- NOTE: dispatches/returns/audit have no update policy (immutable records by design)
