name: Build Winstar Desktop App

on:
  push:
    branches: [ main ]
  workflow_dispatch: # allow manual trigger

jobs:
  build-windows:
    runs-on: windows-latest
    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'

      - name: Install dependencies
        run: npm install

      - name: Build Windows .exe
        run: npm run build:win
        env:
          GH_TOKEN: ${{ secrets.GITHUB_TOKEN }}

      - name: Upload installer (.exe setup)
        uses: actions/upload-artifact@v4
        with:
          name: Winstar-Setup-Windows
          path: dist/*Setup*.exe
          retention-days: 30

      - name: Upload portable (.exe no install)
        uses: actions/upload-artifact@v4
        with:
          name: Winstar-Portable-Windows
          path: dist/*portable*.exe
          retention-days: 30

  # Create a GitHub Release with the .exe attached
  release:
    needs: build-windows
    runs-on: ubuntu-latest
    if: github.ref == 'refs/heads/main'
    steps:
      - name: Download build artifacts
        uses: actions/download-artifact@v4
        with:
          name: Winstar-Portable-Windows
          path: ./release-files

      - name: Create Release
        uses: softprops/action-gh-release@v1
        with:
          tag_name: v2.0.${{ github.run_number }}
          name: Winstar v2.0.${{ github.run_number }}
          body: |
            ## Winstar Desktop App

            ### Download
            - **Winstar-portable.exe** — No installation needed, just run it
            - **Winstar-Setup.exe** — Full installer with desktop shortcut

            ### What's included
            - Full Winstar Uniform Management System
            - Connects to your Supabase cloud database
            - Works offline with local storage fallback
          files: ./release-files/*.exe
          token: ${{ secrets.GITHUB_TOKEN }}
